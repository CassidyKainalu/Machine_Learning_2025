from numpy.linalg import svd
from joblib import Parallel, delayed

np.random.seed(20)
# PCA approximation
def pca_approx(X, k):
    U, S, Vt = svd(X, full_matrices=False)
    Uk = U[:, :k]
    Sk = np.diag(S[:k])
    Vtk = Vt[:k, :]
    return Uk @ Sk @ Vtk

# Cross-validated PCA error
def pca_cv_error(X, k, folds):
    unique_folds = np.unique(folds)
    means = X.mean(axis=0) # column means
    errors = np.zeros(len(unique_folds))
    for idx, f in enumerate(unique_folds): # loop over fold IDs
        X_cv = X.copy() # copy data
        mask = (folds == f)
        for col in range(X.shape[1]): X_cv[mask[:, col], col] = means[col]
        X_hat = pca_approx(X_cv, k) # PCA approx
        sq_err = ((X_hat - X)**2)[mask].sum()
        errors[idx] = sq_err
    return errors.sum() / X.size

# optimal k
optimal_k = np.argmin(errors) + 1
print("Optimal number of PCs:", optimal_k)
